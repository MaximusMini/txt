		[
			'pattern'	=>	'<controller>/<action>/<id:\d+>',	- шаблон, переменная id заменяется её содержимым
			'route'		=>	'<controller>/<action>', 			- маршрут куда передается управление
			'siffix'	=>	'' 									- без суффикса
		]
		
				ругулярное выражение
				\d+ - содержит символы цифр, которых одна и более
